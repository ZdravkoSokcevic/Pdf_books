char *slova(char *s)
{
static char a[80], *pom;
pom = a;
while(*s) {
int uslov = (*s >= 'a' && *s <= 'z') || (*s >= 'A' && *s <= 'Z');
if(uslov)
*pom++ = *s;
s++;
}
*pom = '\0';
return a;
}
int palindrom(char *s)
{
char *s1 = s;
while(*s1)
s1++;
s1--;
while(s < s1)
if(*s++ != *s1--)
return 0;
return 1;
}
