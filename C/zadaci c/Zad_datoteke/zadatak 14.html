void fajl_ispis(char *w)
{
char c;
FILE *fp;
if((fp=fopen(w,"r"))==NULL)
{
printf("Nema fajle imena %s\n",w);
return;
}
while((c=getc(fp))!=EOF)
printf("%c",c);
fclose(fp);printf("\n");
}
void fajl_upis(char *w)
{
char c;
FILE *fp;
if((fp=fopen(w,"w"))==NULL)
{
printf("Ne mogu otvoriti fajlu %s\n",w);
return;
}
printf("Upisuj, za kraj kucaj znak @ :\n");
while((c=getc(stdin))!='@')
putc(c,fp);
fclose(fp);
}
