#include<stdio.h>
#include<conio.h>
#include<stdlib.h>
typedef enum {NETACNO, TACNO} LOGIC;

main()
{
long zn;
FILE *pf;
LOGIC ureci;
long rec_broj;
pf=fopen("test.txt", "r");
rec_broj=0;
ureci=NETACNO;
while((zn=getc(pf))!=EOF)
{
if(zn==' ' || zn=='\t' || zn=='\n')
ureci=NETACNO;
else if(ureci==NETACNO)
{
++rec_broj;
ureci=TACNO;
}
}
printf("%ld\n",rec_broj);
